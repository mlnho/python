# GROUP BY 조건 : HAVING 사용
----------------------------------------------------
SELECT NAME, COUNT(NAME)
FROM ANIMAL_INS
GROUP BY NAME
HAVING NAME IS NOT NULL

# LIMIT : N개의 행만 출력

%D	Month 월(1st, 2dn, 3rd ...)
%Y	Year 연도(1999, 2000, 2020)
%y	Year 연도(99, 00, 20)
%X	Year 연도(1999, 2000, 2020) %V와 같이쓰임
%x	Year 연도(1999, 2000, 2020) %v와 같이쓰임
%a	Day of Week요일(Sun, Mon, Tue ...)

%c	Month(1, 2, 3 ..)

%S, %s	Second 초
